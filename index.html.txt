// IMAGE LOADER — FileReader → base64 → affichage immédiat + sauvegarde
// ============================================================
var imgStore = {}; // key: 'hero','vitrine','p1','p2','p3','np' + new products

function loadImg(input, key) {
  var file = input.files[0];
  if (!file) return;
  setStatus(key, '⏳ Chargement...', '');

  var reader = new FileReader();
  reader.onload = function(e) {
    var b64 = e.target.result;
    imgStore[key] = b64;
    applyImage(key, b64);
    setStatus(key, '✅ Image appliquée !', 'ok');

    // Show preview
    var prev = document.getElementById(key + '-preview');
    if (prev) { prev.src = b64; prev.style.display = 'block'; }
  };
  reader.onerror = function() { setStatus(key, '❌ Erreur de chargement', 'err'); };
  reader.readAsDataURL(file);
}

function applyImage(key, b64) {
  if (key === 'hero') {
    var bg = document.getElementById('hero-bg');
    bg.style.backgroundImage = "url('" + b64 + "')";
    updateHero();
  } else if (key === 'vitrine') {
    var img = document.getElementById('vitrine-img');
    var ph = document.getElementById('vitrine-placeholder');
    img.src = b64;
    img.style.display = 'block';
    if (ph) ph.style.display = 'none';
    updateVitrine();
  } else if (key === 'p1' || key === 'p2' || key === 'p3') {
    var n = key[1];
    var pimg = document.getElementById('prod-img-' + n);
    var pph = document.getElementById('prod-ph-' + n);
    pimg.src = b64;
    pimg.style.display = 'block';
    if (pph) pph.style.display = 'none';
  }
  // 'np' is handled at addProduct time
}

function setStatus(key, msg, cls) {
  var el = document.getElementById(key + '-status');
  if (!el) return;
  el.textContent = msg;
  el.className = 'img-status ' + (cls || '');
}

  document.getElementById('hero-overlay').style.background =

function addProduct() {

function saveAll() {
  try {
    var texts = {};
    TEXT_IDS.forEach(function(id){ var el = document.getElementById(id); if (el) texts[id] = el.textContent; });
    var settings = {
      texts: texts,
      lang: lang,
      heroZoom: document.getElementById('s-zoom').value,
      heroPX: document.getElementById('s-px').value,
      heroPY: document.getElementById('s-py').value,
      heroDark: document.getElementById('s-dark').value,
      vitFit: document.getElementById('vit-fit').value,
      vitPos: document.getElementById('vit-pos').value,
      ig: document.getElementById('ig-link').value,
      wa: document.getElementById('wa-number').value,
      prodRatio: document.getElementById('prod-ratio').value
    };
    localStorage.setItem('gda-settings', JSON.stringify(settings));
    // Save images separately (they can be large)
    var imgKeys = ['hero','vitrine','p1','p2','p3'];
    imgKeys.forEach(function(k){ if (imgStore[k]) { try { localStorage.setItem('gda-img-' + k, imgStore[k]); } catch(e){ console.warn('Image ' + k + ' too large for localStorage'); } } });
    // Save added products
    if (imgStore._products) { try { localStorage.setItem('gda-products', JSON.stringify(imgStore._products)); } catch(e){} }
    alert('✅ Tout a été sauvegardé !');
  } catch(e) { alert('⚠ Erreur de sauvegarde : ' + e.message); }
}

function resetAll() {
  if (confirm('Réinitialiser toutes les modifications ?')) {
    ['gda-settings','gda-img-hero','gda-img-vitrine','gda-img-p1','gda-img-p2','gda-img-p3','gda-products'].forEach(function(k){ localStorage.removeItem(k); });
    location.reload();
  }
}

window.addEventListener('load', function() {
  // Restore images first
  var imgKeys = ['hero','vitrine','p1','p2','p3'];
  imgKeys.forEach(function(k) {
    var b64 = localStorage.getItem('gda-img-' + k);
    if (b64) { imgStore[k] = b64; applyImage(k, b64); var prev = document.getElementById(k + '-preview'); if (prev) { prev.src = b64; prev.style.display = 'block'; } }
  });

  // Restore settings
  var raw = localStorage.getItem('gda-settings');
  if (raw) {
    try {
      var s = JSON.parse(raw);
      if (s.texts) Object.keys(s.texts).forEach(function(id){ var el = document.getElementById(id); if (el) el.textContent = s.texts[id]; });
      if (s.heroZoom) { document.getElementById('s-zoom').value = s.heroZoom; }
      if (s.heroPX) { document.getElementById('s-px').value = s.heroPX; }
      if (s.heroPY) { document.getElementById('s-py').value = s.heroPY; }
      if (s.heroDark) { document.getElementById('s-dark').value = s.heroDark; }
      updateHero(); updateOverlay();
      if (s.vitFit) document.getElementById('vit-fit').value = s.vitFit;
      if (s.vitPos) document.getElementById('vit-pos').value = s.vitPos;
      updateVitrine();
      if (s.ig) { document.getElementById('ig-link').value = s.ig; }
      if (s.wa) { document.getElementById('wa-number').value = s.wa; }
      if (s.ig || s.wa) applySocials();
      if (s.prodRatio) { document.getElementById('prod-ratio').value = s.prodRatio; applyRatio(); }
      if (s.lang && s.lang !== 'fr') { lang = s.lang; applyLang(lang); }
    } catch(e) { console.warn('Load error', e); }
  }

  // Restore added products
  var prods = localStorage.getItem('gda-products');
  if (prods) {
    try {
      JSON.parse(prods).forEach(function(p) {
        var g = document.getElementById('products-grid');
        var c = document.createElement('div'); c.className = 'product-card';
        c.innerHTML = '<div class="product-img-wrapper">' +
          (p.img ? '<img src="' + p.img + '" alt="' + p.name + '" style="width:100%;height:100%;object-fit:cover;display:block">' : '<div class="img-placeholder">📦<br>Photo à venir</div>') +

        g.appendChild(c);
      });
    } catch(e) {}
  }
});
